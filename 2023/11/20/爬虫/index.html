import requests
import csv

headers = {
‘User-Agent’: ‘Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/109.0.0.0 Safari/537.36 SLBrowser/9.0.0.10191 SLBChan/105’
}

url = ‘https://image.baidu.com/search/acjson?tn=resultjson_com&ipn=rj&ct=201326592&is=0%2C0&fp=detail&logid=10088269307996543668&cl=2&lm=-1&ie=utf-8&oe=utf-8&adpicid=0&lpn=0&st=-1&word=LPL%E5%85%A8%E7%90%83%E6%80%BB%E5%86%B3%E8%B5%9B2018ig&z=0&ic=0&hd=&latest=&ctd=1524389385035%5E00_184.222.27.93&ic=0&s=undefined&se=&tab=0&width=&height=&face=undefined&istype=2&qc=&nc=&fr=&simics=&simid=&srctype=&bdtype=0&rpstart=0&rpnum=0&cs=2587220473%2C900917932&catename=&tn7=&isAsyn=1&album_snum=‘

response = requests.get(url, headers=headers)
p_url = response.json()[‘data’]

rows = [[“title”, “url”]]
for i in range(29):
title = p_url[i][‘fromPageTitle’]
url = p_url[i][‘hoverURL’]
row = [title, url]
rows.append(row)
res = requests.get(url=url, headers=headers)
with open(f’百度图片\{title}.jpeg’, mode=’wb’) as f:
f.write(res.content)

with open(‘百度图片.csv’, ‘w’, newline=’’, encoding=’utf-8’) as f:
writer = csv.writer(f)
writer.writerows(rows)

print(‘图片已成功爬取并保存到本地！’)
